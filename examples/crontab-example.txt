# Deepfreeze Rotation Crontab Example
#
# This file provides example crontab entries for running deepfreeze rotation
# on a monthly schedule.
#
# Installation:
#   Option 1: Add to user crontab
#     crontab -e
#     # Paste the desired line below
#
#   Option 2: Add to system crontab
#     sudo vi /etc/cron.d/deepfreeze-rotate
#     # Paste the desired line below (include username field)
#
# Security notes:
#   - Do not paste access keys, passwords, or tokens directly into crontab.
#   - Prefer instance/workload identity, named profiles, or a root-owned env file.
#   - Use absolute paths because cron runs with a minimal environment.

# -----------------------------------------------------------------------------
# OPTION 1: Run on the 1st of every month at 2:00 AM (RECOMMENDED)
# Simple and predictable monthly rotation schedule.
# -----------------------------------------------------------------------------
0 2 1 * * /usr/local/bin/deepfreeze --config /etc/deepfreeze/config.yml rotate --keep 6

# -----------------------------------------------------------------------------
# OPTION 2: Run on the 1st and 15th of every month (approximately every 15 days)
# Use this if you want more frequent rotation.
# -----------------------------------------------------------------------------
# 0 2 1,15 * * /usr/local/bin/deepfreeze --config /etc/deepfreeze/config.yml rotate --keep 6

# -----------------------------------------------------------------------------
# OPTION 3: Run weekly on Sunday at 2:00 AM
# Use this only if your retention policy expects more frequent rotation.
# -----------------------------------------------------------------------------
# 0 2 * * 0 /usr/local/bin/deepfreeze --config /etc/deepfreeze/config.yml rotate --keep 6

# -----------------------------------------------------------------------------
# SYSTEM CRONTAB FORMAT (for /etc/cron.d/deepfreeze-rotate)
# Includes username field between schedule and command.
# -----------------------------------------------------------------------------
# 0 2 1 * * elasticsearch /usr/local/bin/deepfreeze --config /etc/deepfreeze/config.yml rotate --keep 6

# -----------------------------------------------------------------------------
# WITH LOGGING
# Send output to syslog/journald instead of creating ad-hoc world-readable logs.
# -----------------------------------------------------------------------------
# 0 2 1 * * /usr/local/bin/deepfreeze --config /etc/deepfreeze/config.yml rotate --keep 6 2>&1 | /usr/bin/logger -t deepfreeze-rotate

# -----------------------------------------------------------------------------
# WITH ENVIRONMENT VARIABLES
# A named profile is acceptable. Do not store raw secrets in crontab.
# -----------------------------------------------------------------------------
# AWS_PROFILE=default
# 0 2 1 * * /usr/local/bin/deepfreeze --config /etc/deepfreeze/config.yml rotate --keep 6
